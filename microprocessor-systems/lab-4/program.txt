// Variant 1: cyclic right shift (rotate right)
// a = 11 (0x0000000B), sw_i from switches
// Registers:
// r1=a, r2=sw_i, r3=31, r4=k, r5=32, r6=32-k,
// r7=a>>k, r8=a<<(32-k), r9=result

//J B WS  const/ALUop  RA1   RA2   offset     WA
  0 0 00   00000000000000000001011  00001  // r1 = a = 11
  0 0 10   00000000000000000000000  00010  // r2 = sw_i
  0 0 00   00000000000000000011111  00011  // r3 = 31
  0 0 01 00111 00010 00011 00000000 00100  // r4 = r2 & r3 (k = sw_i & 31)
  0 0 00   00000000000000000100000  00101  // r5 = 32
  0 0 01 01000 00101 00100 00000000 00110  // r6 = r5 - r4 (32 - k)
  0 0 01 00101 00001 00100 00000000 00111  // r7 = r1 >> r4
  0 0 01 00001 00001 00110 00000000 01000  // r8 = r1 << r6
  0 0 01 00110 00111 01000 00000000 01001  // r9 = r7 | r8
  1 0 00 00000 01001 00000 00000000 00000  // infinite loop, out_o = r9
